import javax.swing.JPanel;import javax.swing.JFrame;import java.awt.Graphics;import java.awt.Graphics2D;import java.awt.Color;import java.awt.Stroke;import java.awt.BasicStroke;import javax.swing.event.MouseInputAdapter;import java.awt.event.KeyListener;import java.awt.event.MouseEvent;import java.awt.event.KeyEvent;public class drawCircle extends ListenerPanel{  private static int circleX = 100;  private static int circleY = 100;  private static drawCircle panel = new drawCircle();    public void mousePressed (MouseEvent e)    {      circleX = e.getX();      circleY = e.getY();      panel.repaint();    }        public void mouseReleased (MouseEvent e)    {      circleXend = e.getX();      circleYend = e.getY();      panel.repaint();    }   public void paintComponent (Graphics g)   {    Graphics2D g2d = (Graphics2D)g;    g.setColor(Color.white);    g.fillRect( 0,0 , getWidth() , getHeight() );    g.setColor(Color.black);     g.drawOval(circleX,circleY , circleXend , circleYend );  }  public static void main (String[] args) {        JFrame canvas = new JFrame();    canvas.setContentPane( panel );       MyListenerClass listener = new MyListenerClass();    panel.addMouseListener( listener );    panel.addMouseMotionListener ( listener );    panel.addMouseWheelListener ( listener );    canvas.addKeyListener( listener );         canvas.setSize( 800 , 600 );    canvas.setVisible ( true );    canvas.toFront();    canvas.setDefaultCloseOperation( JFrame.EXIT_ON_CLOSE );  }}
